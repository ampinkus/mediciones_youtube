/**
 * @module utils/pathHelper
 * @description Controlador que maneja:
 * Utilidad para obtener `__filename` y `__dirname` en módulos ES (ECMAScript modules).
 * Este módulo proporciona dos constantes que representan la ruta absoluta del archivo actual
 * y del directorio contenedor, similares a las variables `__filename` y `__dirname` de CommonJS.
 */

/*
 * Este archivo se usa para obtener la ruta del directorio actual
 * y evitar repetir código en `app.js`.
 * 
 * 1. Importación de módulos:
 *    - path: Utilidades para trabajar con rutas de archivos y directorios.
 *    - fileURLToPath: Convierte una URL de archivo (file URL) en una ruta de sistema.
 *
 * 2. Obtención del path absoluto del archivo actual:
 *    - import.meta.url devuelve la URL del módulo actual (este archivo JS).
 *    - fileURLToPath convierte esa URL en una ruta de archivo y la asigna a __filename.
 *    - path.dirname obtiene el directorio contenedor de ese archivo y lo asigna a __dirname.
 */

import path from "path";
import { fileURLToPath } from "url";

/**
 * Ruta absoluta del archivo actual.
 * @constant {string}
 */
const __filename = fileURLToPath(import.meta.url);

// console.log(__filename); // la ruta completa del archivo utils 

/**
 * Ruta absoluta del directorio donde se encuentra el archivo actual.
 * @constant {string}
 */
const __dirname = path.dirname(__filename);

// console.log(__dirname);  // el directorio donde está el archivo utils

export { __filename, __dirname };
